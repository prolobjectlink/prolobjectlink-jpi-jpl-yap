542 	@Test
543 	public void testUnify() {
544 
545 		// with atom
546 		PrologAtom atom = provider.newAtom("John Doe");
547 		assertFalse(map.unify(atom));
548 
549 		// with integer
550 		PrologInteger iValue = provider.newInteger(28);
551 		assertFalse(map.unify(iValue));
552 
553 		// with long
554 		PrologLong lValue = provider.newLong(28);
555 		assertFalse(map.unify(lValue));
556 
557 		// with float
558 		PrologFloat fValue = provider.newFloat(36.47);
559 		assertFalse(map.unify(fValue));
560 
561 		// with double
562 		PrologDouble dValue = provider.newDouble(36.47);
563 		assertFalse(map.unify(dValue));
564 
565 		// with variable
566 		PrologVariable variable = provider.newVariable("X", 0);
567 		// true. case predicate and variable
568 		assertTrue(map.unify(variable));
569 
570 		// with predicate
571 		PrologStructure structure1 = provider.parseStructure("some_predicate(X)");
572 		PrologStructure structure2 = provider.parseStructure("some_predicate(28)");
573 
574 		// true because are equals
575 		assertTrue(map.unify(map));
576 		// true because match and their arguments unify
577 		assertFalse(map.unify(structure1));
578 		// false because match but their arguments not unify
579 		assertFalse(map.unify(structure2));
580 
581 		// with list
582 		PrologList flattenList = provider.parseList("['Some Literal']");
583 		PrologList headTailList = provider.parseList("['Some Literal'|[]]");
584 		PrologTerm empty = provider.prologEmpty();
585 		assertFalse(map.unify(flattenList));
586 		assertFalse(map.unify(headTailList));
587 		assertFalse(map.unify(empty));
588 
589 		// with expression
590 		PrologTerm expression = provider.parseTerm("58+93*10");
591 		assertFalse(map.unify(expression));
592 
593 	}
