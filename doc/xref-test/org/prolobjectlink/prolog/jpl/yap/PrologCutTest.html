144 	@Test
145 	public final void testUnify() {
146 
147 		// with atom
148 		PrologTerm cut = provider.prologCut();
149 		PrologAtom atom = provider.newAtom("John Doe");
150 		assertFalse(cut.unify(atom));
151 
152 		// with integer
153 		PrologInteger iValue = provider.newInteger(36);
154 		assertFalse(cut.unify(iValue));
155 
156 		// with long
157 		PrologLong lValue = provider.newLong(28);
158 		assertFalse(cut.unify(lValue));
159 
160 		// with float
161 		PrologFloat fValue = provider.newFloat(36.47);
162 		assertFalse(cut.unify(fValue));
163 
164 		// with double
165 		PrologDouble dValue = provider.newDouble(36.47);
166 		assertFalse(cut.unify(dValue));
167 
168 		// with variable
169 		PrologVariable variable = provider.newVariable("X", 0);
170 		// true. case [] and variable
171 		assertTrue(cut.unify(variable));
172 
173 		// with predicate
174 		PrologStructure structure = provider.parseStructure("some_predicate(a,b,c)");
175 		assertFalse(cut.unify(structure));
176 
177 		// with list
178 		PrologList list = provider.parseList("[a,b,c]");
179 		assertFalse(cut.unify(list));
180 		assertTrue(cut.unify(cut));
181 
182 		// with expression
183 		PrologTerm expression = provider.parseTerm("58+93*10");
184 		assertFalse(cut.unify(expression));
185 
186 	}
